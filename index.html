批量处理示例

# 解密
RESP=$(curl -s -X POST \
  -F "file=@song.ncm" \
  http://localhost:8932/api/convert)
# 下载
ID=$(echo "$RESP" | jq -r .id)
NAME=$(echo "$RESP" | jq -r .output_name)
curl "http://localhost:8932/api/download?id=$ID&name=$NAME" -o "$NAME"
